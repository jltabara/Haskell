\section*{Operaciones lógicas y comparaciones}

\begin{ejer}

Realiza las siguientes operaciones

\begin{itemize}

\item $\mathrm{True} \wedge  \mathrm{False}.\qquad $  (Operador \textbf{\&\&})

\item $\mathrm{True} \wedge  \mathrm{True}$. 

\item $\mathrm{False} \vee  \mathrm{False}\qquad $ (Operador \textbf{||})

\item $\neg \mathrm{False}.\qquad $ (Operador \textbf{not})

\item $\neg (\mathrm{True} \wedge \mathrm{True})$

\item $5 = 5\qquad $ (Operador \textbf{==})

\item $5 = 12$

\item $5 \neq 5\qquad $ (Operador \textbf{/=})

\item  \verb|"hola"  = "hola"|

\item $ 3 < 8\qquad $ (Operador \textbf{<})

\item $5 \leq 5 \qquad$ (Operador \textbf{<=})

\item $ 6 < \verb|"hola"| \qquad $  (Error)

\item $34.8 \geq 2.3\cdot 10^2\qquad $ (Operador \textbf{>=})

\end{itemize}



\end{ejer}
